Operators

API: Math

              -- number -> number
math_abs(x)
math_ceil(x)
math_floor(x)
math_sqrt(x)

-- number, ..., number -> number
math_max(x1, ..., xn)
math_min(x1, ..., xn)

API: String

              -- string -> number
str_len(s)

-- string -> String
str_reverse(s)

-- string -> [string] 
str_split(s, delimiter)
str_split('hi, there', ',')
-- {'hi', ' there'}

-- string -> string
-- end is exclusive
str_substring(s, start, end)

API: Arrays

              -- [T] -> number 
arr_len(a)

-- [T] -> [T]
arr_reverse(a)
arr_sort(a)
array_pop(a)
array_push(a, elem)
arr_set(a, index, elem)

-- [T] -> T 
arr_get(a, index)

API: Tables

              -- {T} -> number
tbl_len(t)

-- {T} -> boolean
tbl_contains(t, key)

-- {T} -> {T} 
tbl_remove(t, key)
tbl_put(t, key, val)

-- {T} -> {T} 
tbl_get(t, key)
